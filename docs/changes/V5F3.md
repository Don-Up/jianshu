## Part 3: Detailed Changes

### user/[username]/page.tsx[modified]

```typescript
// user/[username]/page.tsx
export default function UserProfilePage() {
  const [activeTab, setActiveTab] = useState<'articles' | 'collections'>('articles');
  const { collections, isLoading: isCollectionsLoading } = useCollections();

  return (
    <PageLayout>
      <ProfileHeader user={profile} isOwnProfile={isOwnProfile} />

      <div className="max-w-5xl mx-auto px-4 py-8">
        <div className="flex gap-8">
          <main className="flex-1">
            {/* Tab Navigation */}
            <div className="flex gap-4 border-b mb-6">
              <button
                onClick={() => setActiveTab('articles')}
                className={`pb-3 px-1 text-sm font-medium ${
                  activeTab === 'articles'
                    ? 'text-primary border-b-2 border-primary'
                    : 'text-muted-foreground hover:text-foreground'
                }`}
              >
                文章
              </button>
              {isOwnProfile && (
                <button
                  onClick={() => setActiveTab('collections')}
                  className={`pb-3 px-1 text-sm font-medium ${
                    activeTab === 'collections'
                      ? 'text-primary border-b-2 border-primary'
                      : 'text-muted-foreground hover:text-foreground'
                  }`}
                >
                  收藏集
                </button>
              )}
            </div>

            {/* Tab Content */}
            {activeTab === 'articles' ? (
              <ArticleList articles={articles} />
            ) : (
              <CollectionList
                collections={collections}
                isOwner={isOwnProfile}
                isLoading={isCollectionsLoading}
              />
            )}
          </main>
        </div>
      </div>
    </PageLayout>
  );
}
```

**Description:** Added tab navigation with "文章" and "收藏集" tabs. Collections tab only visible when viewing own profile.

**Steps:**
