## 1. use-search.ts — 搜索状态 Hook

**路径**: `apps/web/src/hooks/use-search.ts`

**职责**: 管理搜索状态和分页逻辑，供搜索结果页使用。

### 核心设计

- **query 驱动**: Hook 通过 `useEffect` 监听 `query` 变化自动触发搜索
- **分页支持**: `loadResults` 方法支持 replace（第一页）和 append（加载更多）两种模式
- **稳定引用**: `loadResults` 用 `useCallback` 包装，避免放入 `useEffect` 依赖导致无限循环

### 关键代码

```typescript
export function useSearch(query: string): UseSearchResult {
  const [articles, setArticles] = useState<ArticleWithAuthor[]>([]);
  const [isLoading, setIsLoading] = useState(false);
  const [hasMore, setHasMore] = useState(false);
  const [page, setPage] = useState(1);
  const [totalPages, setTotalPages] = useState(1);

  const loadResults = useCallback(async (searchQuery: string, pageNum: number = 1) => {
    if (!searchQuery?.trim()) {
      setArticles([]);
      return;
    }

    const res = await articleApi.list({ search: searchQuery, page: pageNum, limit: 20 });

    if (res.success && res.data) {
      // 分页模式判断
      if (pageNum === 1) {
        setArticles(res.data.items);  // 替换
      } else {
        setArticles((prev) => [...prev, ...res.data!.items]);  // 追加
      }
      setTotalPages(res.data.totalPages);
      setHasMore(pageNum < res.data.totalPages);
      setPage(pageNum);
    }
  }, []);

  // query 变化时自动触发搜索（从第1页开始）
  useEffect(() => {
    if (query?.trim()) {
      loadResults(query, 1);
    } else {
      setArticles([]);
    }
  }, [query, loadResults]);

  const loadMore = useCallback(async () => {
    if (!hasMore || isLoading || page >= totalPages) return;
    await loadResults(query, page + 1);
  }, [hasMore, isLoading, page, totalPages, query, loadResults]);

  return { articles, isLoading, error, hasMore, loadMore };
}
```

## 2. search-bar.tsx — 表单提交式搜索框

**路径**: `apps/web/src/components/layout/search-bar.tsx`

**职责**: 简单的表单提交搜索框，用户按 Enter 后跳转到 `/search?q=xxx`。

### 设计原则

1. **URL 可分享** — 提交后才跳转，URL 包含完整搜索词
2. **防止无意义请求** — 不需要每次按键都请求
3. **明确的用户意图** — 用户按 Enter 表示"我要搜索"

### 关键代码

```typescript
export function SearchBar() {
  const [query, setQuery] = useState('');
  const router = useRouter();

  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault();
    if (query.trim()) {
      router.push(`/search?q=${encodeURIComponent(query.trim())}`);
    }
  };

  return (
    <form onSubmit={handleSubmit} className="relative flex items-center">
      <Input
        type="search"
        placeholder="搜索文章..."
        value={query}
        onChange={(e) => setQuery(e.target.value)}
        onKeyDown={handleKeyDown}
        className="w-48 pr-10"
      />
      <Button type="submit" variant="ghost" size="sm" className="absolute right-0 h-full px-3">
        {/* 搜索图标 SVG */}
      </Button>
    </form>
  );
}
```

## 3. search-input.tsx — 实时下拉搜索组件

**路径**: `apps/web/src/components/search/search-input.tsx`

**职责**: 输入时实时显示下拉搜索结果，支持点击跳转。

### 交互设计

| 行为 | 处理 |
|------|------|
| 输入 2+ 字符 | 300ms 防抖后触发搜索 |
| 搜索中 | 显示旋转加载图标 |
| 有结果 | 下拉显示最多 5 条 |
| 点击结果 | 跳转到文章页，关闭下拉，清空输入 |
| 点击外部 | 关闭下拉 |
| Escape 键 | 关闭下拉 |
| 无结果 | 显示"未找到相关文章" |

### 关键代码

```typescript
export function SearchInput({ className }: SearchInputProps) {
  const [query, setQuery] = useState('');
  const [results, setResults] = useState<ArticleWithAuthor[]>([]);
  const [isLoading, setIsLoading] = useState(false);
  const [isOpen, setIsOpen] = useState(false);
  const inputRef = useRef<HTMLInputElement>(null);
  const dropdownRef = useRef<HTMLDivElement>(null);
  const router = useRouter();

  // 搜索函数
  const searchArticles = useCallback(async (searchQuery: string) => {
    if (searchQuery.trim().length < 2) {
      setResults([]);
      setIsOpen(false);
      return;
    }

    setIsLoading(true);
    try {
      const res = await articleApi.list({ search: searchQuery, page: 1, limit: 5 });
      if (res.success && res.data) {
        setResults(res.data.items);
        setIsOpen(true);
      }
    } finally {
      setIsLoading(false);
    }
  }, []);

  // 300ms 防抖
  useEffect(() => {
    const timer = setTimeout(() => searchArticles(query), 300);
    return () => clearTimeout(timer);
  }, [query, searchArticles]);

  // 点击外部关闭下拉
  useEffect(() => {
    const handleClickOutside = (event: MouseEvent) => {
      if (
        dropdownRef.current && !dropdownRef.current.contains(event.target as Node) &&
        inputRef.current && !inputRef.current.contains(event.target as Node)
      ) {
        setIsOpen(false);
      }
    };
    document.addEventListener('mousedown', handleClickOutside);
    return () => document.removeEventListener('mousedown', handleClickOutside);
  }, []);

  const handleResultClick = (slug: string) => {
    setIsOpen(false);
    setQuery('');
    router.push(`/article/${slug}`);
  };

  return (
    <div className={cn('relative', className)}>
      {/* 输入框 + 搜索/加载图标 */}
      <div className="relative flex items-center">
        <Input ref={inputRef} type="search" placeholder="搜索文章..." value={query}
          onChange={(e) => setQuery(e.target.value)} className="w-48 pr-10" />
        <Button variant="ghost" size="sm" className="absolute right-0 h-full px-3" disabled>
          {isLoading ? <SpinnerIcon /> : <SearchIcon />}
        </Button>
      </div>

      {/* 下拉结果 */}
      {isOpen && (
        <div ref={dropdownRef} className="absolute right-0 top-full mt-1 w-80 rounded-md border bg-background shadow-lg z-50">
          {results.length > 0 ? (
            <ul className="py-1 max-h-96 overflow-y-auto">
              {results.map((article) => (
                <li key={article.id}>
                  <button type="button" onClick={() => handleResultClick(article.slug)}
                    className="w-full text-left px-4 py-3 hover:bg-muted flex items-start gap-3">
                    <Avatar className="h-8 w-8 mt-0.5">
                      <AvatarImage src={article.author.avatar || undefined} />
                      <AvatarFallback>{article.author.name.slice(0, 2).toUpperCase()}</AvatarFallback>
                    </Avatar>
                    <div className="flex-1 min-w-0">
                      <p className="text-sm font-medium truncate">{article.title}</p>
                      <p className="text-xs text-muted-foreground">
                        {article.author.name} · {formatDate(article.createdAt)}
                      </p>
                    </div>
                  </button>
                </li>
              ))}
            </ul>
          ) : (
            <div className="px-4 py-6 text-center text-sm text-muted-foreground">未找到相关文章</div>
          )}
        </div>
      )}
    </div>
  );
}
```

## 4. search/page.tsx — 搜索结果页

**路径**: `apps/web/src/app/search/page.tsx`

**职责**: 显示搜索结果列表，支持分页加载。

### 关键设计

- **Suspense 边界**: `useSearchParams()` 必须包裹在 `<Suspense>` 内
- **复用 ArticleList**: 搜索结果和首页使用同一个列表组件
- **状态处理**: 空关键词、加载中、错误、无结果三种状态

### 关键代码

```typescript
function SearchContent() {
  const searchParams = useSearchParams();
  const query = searchParams.get('q') || '';
  const { articles, isLoading, error, hasMore, loadMore } = useSearch(query);

  if (!query) return <div className="text-center py-12"><p>请输入搜索关键词</p></div>;
  if (error) return <div className="text-center py-12"><p className="text-red-500">搜索失败: {error}</p></div>;
  if (!isLoading && articles.length === 0) return <div className="text-center py-12"><p>没有找到相关文章</p></div>;

  return (
    <div className="space-y-4">
      <ArticleList articles={articles} isLoading={isLoading} />
      {hasMore && <Button variant="outline" onClick={loadMore}>{isLoading ? '加载中...' : '加载更多'}</Button>}
    </div>
  );
}

export default function SearchPage() {
  return (
    <PageLayout>
      <div className="mx-auto max-w-5xl px-4 py-8">
        <h1 className="text-2xl font-bold mb-8">搜索结果</h1>
        <Suspense fallback={<SearchFallback />}>
          <SearchContent />
        </Suspense>
      </div>
    </PageLayout>
  );
}
```

## 数据流

### 基础搜索（SearchBar → 搜索结果页）

```
用户输入关键词
       ↓
SearchBar 输入框 onChange → 更新本地 state
       ↓
用户按 Enter / 点击搜索图标
       ↓
router.push(/search?q=xxx) → URL 变化
       ↓
SearchPage 挂载 → useSearchParams() 读取 ?q=
       ↓
useSearch(query) → articleApi.list({ search: query })
       ↓
API 返回 → setArticles → ArticleList 渲染
```

### 实时下拉搜索（SearchInput）

```
用户输入关键词
       ↓
300ms 防抖
       ↓
articleApi.list({ search: query, limit: 5 })
       ↓
返回结果 → 显示下拉
       ↓
用户点击结果 → router.push(/article/{slug})
   或 点击外部 / Escape → 关闭下拉
```

## 注意事项

1. **API 调用必须包含 `page` 字段**: `ArticleListParams` 继承自 `PaginationParams`，需要 `page` 和 `limit` 字段
2. **防抖阈值 300ms**: 平衡了响应速度和 API 调用次数
3. **最小搜索字符 2**: 避免过短的搜索词产生过多结果
4. **下拉结果限制 5 条**: 实时下拉不需要太多结果，主要用于快速导航
5. **Result 页面限制 20 条/页**: 与首页文章列表一致
